cmake_minimum_required(VERSION 3.10)


project(caesar_cipher

	VERSION 1.0
	DESCRIPTION "A cmake file for caesar cipher."
	LANGUAGES C)

set(SOURCES
	src/caesar_decrypt.c
	src/caesar_encrypt.c
)

add_library(caesar STATIC ${SOURCES})

target_include_directories(
        caesar PUBLIC  ${CMAKE_CURRENT_SOURCE_DIR}/inc
)

target_compile_options(caesar PRIVATE -Wall -Wextra -Wpedantic)
